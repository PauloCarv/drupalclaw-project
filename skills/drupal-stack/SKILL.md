# drupal-stack

Manages the Docker container stack for Drupal development.

## Parameters

- `action` — start | stop | status | restart | destroy (default: start)
- `db` — mariadb | postgres | sqlite (default: mariadb)
- `php_port` — port for nginx/PHP (default: auto-detect)
- `db_port` — port for DB (default: auto-detect)

## Steps

### 1. Check Docker socket

```bash
if ! docker info >/dev/null 2>&1; then
  echo "❌ Docker socket not available."
  echo "Make sure the PiClaw container was started with:"
  echo "  -v /var/run/docker.sock:/var/run/docker.sock"
  exit 1
fi
echo "✅ Docker available: $(docker --version)"
```

### 2. Set variables and detect host path

```bash
WORKSPACE_DIR="/workspace"
DB_TYPE="${DB_TYPE:-mariadb}"

# Generate or load a stable workspace ID — used as docker compose project name
# so containers from different workspaces never collide.
WORKSPACE_ID_FILE="${WORKSPACE_DIR}/.piclaw/workspace-id"
mkdir -p "${WORKSPACE_DIR}/.piclaw"
if [[ -f "$WORKSPACE_ID_FILE" ]]; then
  WORKSPACE_ID=$(cat "$WORKSPACE_ID_FILE")
else
  WORKSPACE_ID=$(cat /proc/sys/kernel/random/uuid 2>/dev/null | tr -d '-' | cut -c1-8)
  echo "$WORKSPACE_ID" > "$WORKSPACE_ID_FILE"
fi
PROJECT_NAME="drupal-${WORKSPACE_ID}"
echo "🔑 Workspace: ${PROJECT_NAME}"

# docker compose talks to the HOST daemon — volume paths must be
# host paths, not container paths. Detected via docker inspect on this container.
SELF_ID=$(cat /etc/hostname 2>/dev/null || hostname)
HOST_WORKSPACE=$(docker inspect "$SELF_ID" --format '{{range .Mounts}}{{if eq .Destination "/workspace"}}{{.Source}}{{end}}{{end}}' 2>/dev/null)

if [[ -z "$HOST_WORKSPACE" ]]; then
  # Fallback: try by container name
  SELF_NAME=$(docker ps --filter "id=$SELF_ID" --format '{{.Names}}' 2>/dev/null | head -1)
  [[ -n "$SELF_NAME" ]] && HOST_WORKSPACE=$(docker inspect "$SELF_NAME" --format '{{range .Mounts}}{{if eq .Destination "/workspace"}}{{.Source}}{{end}}{{end}}' 2>/dev/null)
fi

if [[ -z "$HOST_WORKSPACE" ]]; then
  echo "❌ Cannot determine the host path for /workspace."
  echo "   Check that the container has the /workspace volume mounted."
  exit 1
fi
echo "🗂️  Host workspace: $HOST_WORKSPACE"

STACK_DIR="${WORKSPACE_DIR}/.piclaw/stack"
# COMPOSE_FILE uses HOST path so docker compose can find it
COMPOSE_FILE="${HOST_WORKSPACE}/docker-compose.drupal.yml"

DRUPAL_DIR="${HOST_WORKSPACE}/drupal"
if [[ ! -d "${WORKSPACE_DIR}/drupal" ]]; then
  DRUPAL_DIR="${HOST_WORKSPACE}"
fi
```

### 3. Detect available ports

```bash
find_free_port() {
  local start_port=$1
  local port=$start_port
  while [ $port -lt $((start_port + 100)) ]; do
    if ! docker ps --format '{{.Ports}}' | grep -q ":${port}->" && \
       ! ss -tlnp 2>/dev/null | grep -q ":${port} " && \
       ! lsof -i :${port} >/dev/null 2>&1; then
      echo $port
      return 0
    fi
    port=$((port + 1))
  done
  echo ""
  return 1
}

if [[ -z "${PHP_PORT:-}" ]]; then
  PHP_PORT=$(find_free_port 8085)
  if [[ -z "$PHP_PORT" ]]; then
    echo "❌ No free port found for PHP (tried 8085-8184)"
    exit 1
  fi
fi
echo "📦 PHP/nginx port: $PHP_PORT"

if [[ "$DB_TYPE" != "sqlite" ]]; then
  DEFAULT_DB_PORT=3306
  [[ "$DB_TYPE" == "postgres" ]] && DEFAULT_DB_PORT=5432
  if [[ -z "${DB_PORT:-}" ]]; then
    DB_PORT=$(find_free_port $DEFAULT_DB_PORT)
    if [[ -z "$DB_PORT" ]]; then
      echo "❌ No free port found for DB"
      exit 1
    fi
  fi
  echo "🗄️  DB port: $DB_PORT ($DB_TYPE)"
else
  DB_PORT=""
  echo "🗄️  DB: SQLite (no DB container)"
fi
```

### 4. Create stack directory

```bash
mkdir -p "$STACK_DIR"
```

### 5. Copy templates

```bash
cp /home/agent/.pi/templates/Dockerfile.php "$STACK_DIR/Dockerfile.php"
cp /home/agent/.pi/templates/nginx.conf "$STACK_DIR/nginx.conf"
```

### 6. Generate docker-compose.drupal.yml

All volume paths use `$HOST_WORKSPACE` (host path) so the Docker daemon resolves them correctly.

```bash
cat > "${WORKSPACE_DIR}/docker-compose.drupal.yml" << 'COMPOSE_HEADER'
# Auto-generated by drupal-stack skill. Do not edit manually.
COMPOSE_HEADER

if [[ "$DB_TYPE" == "mariadb" ]]; then
cat >> "${WORKSPACE_DIR}/docker-compose.drupal.yml" << EOF
services:
  php:
    image: drupal-claw-php:latest
    build:
      context: ${HOST_WORKSPACE}
      dockerfile: ${HOST_WORKSPACE}/.piclaw/stack/Dockerfile.php
    volumes:
      - ${HOST_WORKSPACE}/drupal:/var/www/html
    networks:
      - drupal-net
    depends_on:
      db:
        condition: service_healthy
    environment:
      - DB_HOST=db
      - DB_PORT=3306
      - DB_NAME=drupal
      - DB_USER=drupal
      - DB_PASSWORD=drupal
      - DB_DRIVER=mysql

  nginx:
    image: nginx:alpine
    ports:
      - "${PHP_PORT}:80"
    volumes:
      - ${HOST_WORKSPACE}/drupal:/var/www/html
      - ${HOST_WORKSPACE}/.piclaw/stack/nginx.conf:/etc/nginx/conf.d/default.conf:ro
    networks:
      - drupal-net
    depends_on:
      - php

  db:
    image: mariadb:11
    ports:
      - "${DB_PORT}:3306"
    environment:
      MARIADB_ROOT_PASSWORD: root
      MARIADB_DATABASE: drupal
      MARIADB_USER: drupal
      MARIADB_PASSWORD: drupal
    volumes:
      - drupal_db_data:/var/lib/mysql
    networks:
      - drupal-net
    healthcheck:
      test: ["CMD", "healthcheck.sh", "--connect", "--innodb_initialized"]
      interval: 5s
      timeout: 5s
      retries: 10

networks:
  drupal-net:
    name: ${PROJECT_NAME}_drupal-net

volumes:
  drupal_db_data:
EOF

elif [[ "$DB_TYPE" == "postgres" ]]; then
cat >> "${WORKSPACE_DIR}/docker-compose.drupal.yml" << EOF
services:
  php:
    image: drupal-claw-php:latest
    build:
      context: ${HOST_WORKSPACE}
      dockerfile: ${HOST_WORKSPACE}/.piclaw/stack/Dockerfile.php
    volumes:
      - ${HOST_WORKSPACE}/drupal:/var/www/html
    networks:
      - drupal-net
    depends_on:
      db:
        condition: service_healthy
    environment:
      - DB_HOST=db
      - DB_PORT=5432
      - DB_NAME=drupal
      - DB_USER=drupal
      - DB_PASSWORD=drupal
      - DB_DRIVER=pgsql

  nginx:
    image: nginx:alpine
    ports:
      - "${PHP_PORT}:80"
    volumes:
      - ${HOST_WORKSPACE}/drupal:/var/www/html
      - ${HOST_WORKSPACE}/.piclaw/stack/nginx.conf:/etc/nginx/conf.d/default.conf:ro
    networks:
      - drupal-net
    depends_on:
      - php

  db:
    image: postgres:16-alpine
    ports:
      - "${DB_PORT}:5432"
    environment:
      POSTGRES_DB: drupal
      POSTGRES_USER: drupal
      POSTGRES_PASSWORD: drupal
    volumes:
      - drupal_db_data:/var/lib/postgresql/data
    networks:
      - drupal-net
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U drupal -d drupal"]
      interval: 5s
      timeout: 5s
      retries: 10

networks:
  drupal-net:
    name: ${PROJECT_NAME}_drupal-net

volumes:
  drupal_db_data:
EOF

elif [[ "$DB_TYPE" == "sqlite" ]]; then
cat >> "${WORKSPACE_DIR}/docker-compose.drupal.yml" << EOF
services:
  php:
    image: drupal-claw-php:latest
    build:
      context: ${HOST_WORKSPACE}
      dockerfile: ${HOST_WORKSPACE}/.piclaw/stack/Dockerfile.php
    volumes:
      - ${HOST_WORKSPACE}/drupal:/var/www/html
    networks:
      - drupal-net
    environment:
      - DB_DRIVER=sqlite
      - DB_NAME=/var/www/html/sites/default/files/.sqlite

  nginx:
    image: nginx:alpine
    ports:
      - "${PHP_PORT}:80"
    volumes:
      - ${HOST_WORKSPACE}/drupal:/var/www/html
      - ${HOST_WORKSPACE}/.piclaw/stack/nginx.conf:/etc/nginx/conf.d/default.conf:ro
    networks:
      - drupal-net
    depends_on:
      - php

networks:
  drupal-net:
    name: ${PROJECT_NAME}_drupal-net
EOF
fi

echo "✅ docker-compose.drupal.yml generated"
```

### 7. Action: start

```bash
if [[ "${ACTION:-start}" == "start" ]]; then
  PHP_IMAGE="drupal-claw-php:latest"
  REBUILD_FLAG="--build"
  if docker image inspect "$PHP_IMAGE" >/dev/null 2>&1; then
    IMAGE_AGE=$(docker image inspect "$PHP_IMAGE" --format '{{.Created}}' 2>/dev/null | cut -c1-10)
    echo "ℹ️  Image $PHP_IMAGE already exists (built: $IMAGE_AGE)."
  fi
fi
```

If the image already exists, ask the user:

> The `drupal-claw-php` image was previously built.
> Do you want to do a **new build** or use the existing image?
> - `new build` — rebuilds the image (slower, picks up Dockerfile changes)
> - `use existing` — starts directly with the current image (faster)

```bash
if [[ "${ACTION:-start}" == "start" ]]; then
  # REBUILD_FLAG: "new build" → --build | "use existing" → ""
  # If image did not exist → --build without asking

  docker compose -f "$COMPOSE_FILE" -p "$PROJECT_NAME" up -d $REBUILD_FLAG

  sleep 3

  INTERNAL_URL="http://host.docker.internal:${PHP_PORT}"
  HTTP_CODE=$(curl -s -o /dev/null -w "%{http_code}" --max-time 5 "$INTERNAL_URL" 2>/dev/null || echo "000")
  if [[ "$HTTP_CODE" == "000" ]]; then
    NGINX_CONTAINER=$(docker ps --filter "name=${PROJECT_NAME}" --filter "name=nginx" --format '{{.Names}}' | head -1)
    if [[ -n "$NGINX_CONTAINER" ]]; then
      NGINX_IP=$(docker inspect "$NGINX_CONTAINER" --format '{{range .NetworkSettings.Networks}}{{.IPAddress}}{{end}}' | head -1)
      INTERNAL_URL="http://${NGINX_IP}:80"
      HTTP_CODE=$(curl -s -o /dev/null -w "%{http_code}" --max-time 5 "$INTERNAL_URL" 2>/dev/null || echo "000")
    fi
  fi

  echo ""
  echo "═══════════════════════════════════════════════"
  echo "✅ Drupal stack started!"
  echo "  🌐 Drupal (browser): http://localhost:${PHP_PORT}"
  echo "  🔗 Drupal (internal): ${INTERNAL_URL}"
  [[ -n "$DB_PORT" ]] && echo "  🗄️  DB ($DB_TYPE): localhost:${DB_PORT}"
  echo "  📁 Files: ${DRUPAL_DIR}"
  if [[ "$HTTP_CODE" =~ ^[23] ]]; then
    echo "  ✅ Health check OK (HTTP $HTTP_CODE)"
  else
    echo "  ⚠️  Health check: HTTP $HTTP_CODE — may still be starting"
  fi
  echo "═══════════════════════════════════════════════"
fi
```

### 8. Action: stop

```bash
if [[ "${ACTION:-}" == "stop" ]]; then
  docker compose -f "$COMPOSE_FILE" -p "$PROJECT_NAME" stop
  echo "⏹ Stack stopped. Data preserved."
fi
```

### 9. Action: status

```bash
if [[ "${ACTION:-}" == "status" ]]; then
  docker compose -f "$COMPOSE_FILE" -p "$PROJECT_NAME" ps
fi
```

### 10. Action: restart

```bash
if [[ "${ACTION:-}" == "restart" ]]; then
  docker compose -f "$COMPOSE_FILE" -p "$PROJECT_NAME" restart
  echo "🔄 Stack restarted."
fi
```

### 11. Action: destroy

```bash
if [[ "${ACTION:-}" == "destroy" ]]; then
  docker compose -f "$COMPOSE_FILE" -p "$PROJECT_NAME" down -v
  echo "🗑️ Stack destroyed (volumes removed)."
fi
```

### 12. Save state

```bash
cat > "${WORKSPACE_DIR}/.piclaw/stack/state.json" << EOF
{
  "workspace_id": "$WORKSPACE_ID",
  "project_name": "$PROJECT_NAME",
  "db_type": "$DB_TYPE",
  "php_port": $PHP_PORT,
  "db_port": ${DB_PORT:-null},
  "drupal_url": "http://localhost:${PHP_PORT}",
  "internal_url": "http://host.docker.internal:${PHP_PORT}",
  "started_at": "$(date -Iseconds)"
}
EOF
echo "💾 State saved to .piclaw/stack/state.json"
```
